async function submitMessage(text) {
  isThinking = true;
  sendBtn.disabled = true;
  appendUserMsg(text);

  history.push({ role:'user', content: text });

  showTyping();
  thinkBar.classList.add('active');

  try {
    const resp = await fetch(API_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: MODEL,
        max_tokens: 1000,
        system: SYSTEM,
        messages: history
      })
    });

    const data = await resp.json();
    hideTyping();
    thinkBar.classList.remove('active');

    const reply = data?.content?.map(b => b.text || '').join('') || "Sorry, I couldn't get a response.";
    history.push({ role:'assistant', content: reply });

    // keep last 20 turns
    if (history.length > 40) history = history.slice(-40);

    appendBotMsg(formatReply(reply));
  } catch(err) {
    hideTyping();
    thinkBar.classList.remove('active');
    appendBotMsg("⚠️ Connection error. Please check your API key or try again.");
    console.error(err);
  }

  isThinking = false;
  sendBtn.disabled = false;
  input.focus();
}

function formatReply(text) {
  return text
    .replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')
    .replace(/\*\*(.*?)\*\*/g,'<strong>$1</strong>')
    .replace(/\*(.*?)\*/g,'<em>$1</em>')
    .replace(/`([^`]+)`/g,'<code style="background:var(--surface3);padding:1px 5px;border-radius:4px;font-size:12px">$1</code>')
    .replace(/\n/g,'<br>');
}
